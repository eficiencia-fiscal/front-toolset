(constant) EventBus

Cria um EventBus para poder transmitir eventos entre os frameworks do front-end
Source:
  • front-toolset.js, line 16
Example
import EventBus from './helpers';

EventBus.$emit('EVENT_NAME', function_to_handle); // For publish/emit one event
EventBus.$on('EVENT_NAME', function_to_handle); // For listen/subscribe one event
EventBus.$off('EVENT_NAME', function_to_handle); // For cancel listen/subscribe from event
